NODENAME	SPU [Select Bits]
selb 	"rt, ra, rb, rc"	Select Bits 	rt.bi[n] = (rc.bi[n] == 0) ? ra.bi[n] : rb.bi[n]
gb 	"rt, ra"	Gather Bits from Words 	rt=0;for(n=0;n<4;n++){rt.w[0]|=(rb.w[n]&1);rt.w[0]<<=1}
gbh 	"rt, ra"	Gather Bits from Halfwords 	rt=0;for(n=0;n<8;n++){rt.w[0]|=(rb.h[n]&1);rt.w[0]<<=1}
gbb 	"rt, ra"	Gather Bits from Bytes 	rt=0;for(n=0;n<16;n++){rt.w[0]|=(rb.b[n]&1);rt.w[0]<<=1}
fsm 	"rt, ra"	Form Select Mask for Words 	rt.w[n] = ((ra.b[3] << n) & 0x8) ? 0xF... : 0x0…
fsmh 	"rt, ra"	Form Select Mask for Halfwords 	rt.h[n] = ((ra.b[3] << n) & 0x80) ? 0xF… : 0x0…
fsmb 	"rt, ra"	Form Select Mask for Bytes 	rt.b[n] = ((ra.h[1] << n) & 0x8000) ? 0xF… : 0x0…
fsmbi 	"rt, u16"	Form Select Mask for Bytes Immediate 	rt.b[n] = ((u16 << n) & 0x8000) ? 0xF… : 0x0…

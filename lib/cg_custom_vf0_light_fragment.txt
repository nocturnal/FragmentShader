PREFIX CustomVF0 Fragment Lighting
PREFIX_NODE_NAME custom_vf0_fragment_lighting

# --------
NAME Texture Sampler
NODE_NAME texture_sampler
TYPE IN
VARTYPE sampler2D
VARSEL decal0 decal1 decal2
HELP Access textures (decal0-decal2)

# --------
NAME UV
NODE_NAME uv
TYPE IN
VARTYPE float2
VARSEL float2(uv.xy)
HELP Texture coordinates

# --------
NAME Position
NODE_NAME position
TYPE IN
VARTYPE float4
VARSEL position
HELP Projected position

# --------
NAME Vertex Arguments
NODE_NAME vertex_arguments
TYPE IN
VARTYPE float4
VARSEL arg0 arg1 arg2
HELP Values passed in from the vertex setup fragment

# --------
NAME Normal Scale
NODE_NAME normal_scale
TYPE IN
VARTYPE float
VARSEL float(normal_scale.x)
HELP Normal scale set in shader

# --------
NAME Light Color
NODE_NAME light_color
TYPE IN
VARTYPE float4
VARSEL light_color[i]
HELP The color of the current light

# --------
NAME Lighting Coefficients
NODE_NAME lighting_coefficients
TYPE IN
VARTYPE float4
VARSEL c0 c1 c2 c3 c4
HELP Per light values assigned in fragment setup

# --------
NAME Vertex Light Argument
NODE_NAME vertex_light_argument
TYPE IN
VARTYPE float4
VARSEL light_arg[i]
HELP Per light value assigned in the vertex light fragment

# --------
NAME Final Light Color
NODE_NAME final_light_color
TYPE OUT
VARTYPE float4
VARSEL final_light_color
HELP Assigns the (additive color) contribution by the current light
